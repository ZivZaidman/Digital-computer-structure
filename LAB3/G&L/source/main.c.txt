  while(1){
	switch(state){
	  case state0: //idle - Sleep
	      enterLPM(mode0);
	      break;
	  case state1: //PB0 recorder
	     // Perform Notes recording - Task 1
	      _bis_SR_register(LPM0_bits + GIE);       // CPU off
	    break;

	  case state2: //PB1 Audio player
	      //audio_player(); // Perform Song player - Task

	      break;
		
      case state3: ; // PB2/3 For final lab

		break;

      case state4: //
		break;
		
	}
  }
}
